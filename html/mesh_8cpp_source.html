205 TriangleData Mesh::triangle( const unsigned int id ) const
206 {
207  unsigned int a, b, c;
208  if(m_indices.size() > 0)
209  {
210  assert((size_t) id*3+2 < m_indices.size());
211  a= m_indices[id*3];
212  b= m_indices[id*3 +1];
213  c= m_indices[id*3 +2];
214  }
215  else
216  {
217  assert((size_t) id*3+2 < m_positions.size());
218  a= id*3;
219  b= id*3 +1;
220  c= id*3 +2;
221  }
222 
223  TriangleData triangle;
224  triangle.a= m_positions[a];
225  triangle.b= m_positions[b];
226  triangle.c= m_positions[c];
227 
228  if(m_normals.size() == m_positions.size())
229  {
230  triangle.na= m_normals[a];
231  triangle.nb= m_normals[b];
232  triangle.nc= m_normals[c];
233  }
234  else
235  {
236  // calculer la normale geometrique
237 //#if (_MSC_VER>1500) // je ne sais pas pourquoi mais Visual ne veut pas la 2e version : il veut un truc constant en 2e paramètre mais c'est bien un truc constant
238  Point pb(m_positions[b]);
239  Point pc(m_positions[c]);
240  Vector ab(Point(m_positions[a]), pb);
241  Vector ac(Point(m_positions[a]), pc);
242 //#else
243  // Vector ab(Point(m_positions[a]), Point(m_positions[b]));
244  // Vector ac(Point(m_positions[a]), Point(m_positions[c]));
245 //#endif
246  Vector n= normalize(cross(ab, ac));
247  triangle.na= vec3(n);
248  triangle.nb= vec3(n);
249  triangle.nc= vec3(n);
250  }
251 
252  if(m_texcoords.size() == m_positions.size())
253  {
254  triangle.ta= m_texcoords[a];
255  triangle.tb= m_texcoords[b];
256  triangle.tc= m_texcoords[c];
257  }
258  else
259  {
260  // coordonnees barycentriques des sommets, convention p(u, v)= w*a + u*b + v*c, avec w= 1 - u -v
261  triangle.ta= vec2(0, 0);
262  triangle.tb= vec2(1, 0);
263  triangle.tc= vec2(0, 1);
264  }
265 
266  return triangle;
267 }

312 GLuint Mesh::create_buffers( const bool use_texcoord, const bool use_normal, const bool use_color )
313 {
314  if(m_positions.size() == 0)
315  return 0;
316 
317  // ne creer que les buffers necessaires
318  GLuint vao= create_vertex_format();
319  make_vertex_buffer(vao, 0, 3, GL_FLOAT, vertex_buffer_size(), vertex_buffer());
320 
321  if(m_indices.size() > 0)
322  make_index_buffer(vao, index_buffer_size(), index_buffer());
323 
324 #if 1
325  if(m_texcoords.size() > 0 && m_texcoords.size() < m_positions.size() && use_texcoord)
326  printf("[error] invalid texcoords array...\n");
327  if(m_normals.size() > 0 && m_normals.size() < m_positions.size() && use_normal)
328  printf("[error] invalid normals array...\n");
329  if(m_colors.size() > 0 && m_colors.size() < m_positions.size() && use_color)
330  printf("[error] invalid colors array...\n");
331 #endif
332 
333  if(m_texcoords.size() == m_positions.size() && use_texcoord)
334  make_vertex_buffer(vao, 1, 2, GL_FLOAT, texcoord_buffer_size(), texcoord_buffer());
335  if(m_normals.size() == m_positions.size() && use_normal)
336  make_vertex_buffer(vao, 2, 3, GL_FLOAT, normal_buffer_size(), normal_buffer());
337  if(m_colors.size() == m_positions.size() && use_color)
338  make_vertex_buffer(vao, 3, 4, GL_FLOAT, color_buffer_size(), color_buffer());
339 
340  m_update_buffers= false;
341  return vao;
342 }

344 int Mesh::update_buffers( const bool use_texcoord, const bool use_normal, const bool use_color )
345 {

351  update_vertex_buffer(m_vao, 0, vertex_buffer_size(), vertex_buffer());
352 
353  // ne modifier que les attributs des sommets, pas la topologie / structure du maillage
354  if(m_texcoords.size() == m_positions.size() && use_texcoord)
355  update_vertex_buffer(m_vao, 1, texcoord_buffer_size(), texcoord_buffer());
356  if(m_normals.size() == m_positions.size() && use_normal)
357  update_vertex_buffer(m_vao, 2, normal_buffer_size(), normal_buffer());
358  if(m_colors.size() == m_positions.size() && use_color)
359  update_vertex_buffer(m_vao, 3, color_buffer_size(), color_buffer());
360 
361  m_update_buffers= false;
362  return 1;
363 }

366 GLuint Mesh::create_program( const bool use_texcoord, const bool use_normal, const bool use_color, const bool use_light, const bool use_alpha_test )
367 {
368  std::string definitions;
369 
370  if(m_texcoords.size() == m_positions.size() && use_texcoord)

372  if(m_normals.size() == m_positions.size() && use_normal)

384  return read_program( smart_path("data/shaders/mesh.glsl"), definitions.c_str());

390 void Mesh::draw( const Transform& model, const Transform& view, const Transform& projection,
391  const bool use_light, const Point& light, const Color& light_color,
392  const bool use_texture, const GLuint texture,
393  const bool use_alpha_test, const float alpha_min )
394 {
395  bool use_texcoord= (m_texcoords.size() == m_positions.size() && texture > 0);
396  bool use_normal= (m_normals.size() == m_positions.size());
397  bool use_color= (m_colors.size() == m_positions.size());
398 
399  if(m_vao == 0)
400  // force la creation de tous les buffers
401  m_vao= create_buffers(true, true, true);
402  if(m_update_buffers)
403  update_buffers(true, true, true);
404 
405  unsigned int key= 0;
406  if(use_texcoord) key= key | 1;
407  if(use_normal) key= key | 2;
408  if(use_color) key= key | 4;
409  if(use_texture) key= key | 8;
410  if(use_light) key= key | 16;
411  if(use_alpha_test) key= key | 32;
412 
413  if(m_state != key)
414  // recherche un shader deja compile pour ce type de draw
415  m_program= m_state_map[key];
416 
417  if(m_program == 0)
418  {
419  // pas de shader pour ce type de draw
420  m_program= create_program(use_texcoord, use_normal, use_color, use_light, use_alpha_test);
421  program_print_errors(m_program);
422 
423  // conserver le shader
424  m_state_map[key]= m_program;
425  }
426 
427  // conserve la config du shader selectionne.
428  m_state= key;
429 
430  glBindVertexArray(m_vao);
431  glUseProgram(m_program);
432 
433  program_uniform(m_program, "mesh_color", default_color());
434 
435  Transform mv= view * model;
436  Transform mvp= projection * view * model;
437 
438  program_uniform(m_program, "mvpMatrix", mvp);
439  program_uniform(m_program, "mvMatrix", mv);
440  if(use_normal)
441  program_uniform(m_program, "normalMatrix", mv.normal()); // transforme les normales dans le repere camera.
442 
443  // utiliser une texture, elle ne sera visible que si le mesh a des texcoords...
444  if(texture && use_texcoord && use_texture)
445  program_use_texture(m_program, "diffuse_color", 0, texture);
446 
447  if(use_light)
448  {
449  program_uniform(m_program, "light", view(light)); // transforme la position de la source dans le repere camera, comme les normales
450  program_uniform(m_program, "light_color", light_color);
451  }
452 
453  if(use_alpha_test)
454  program_uniform(m_program, "alpha_min", alpha_min);
455 
456  if(m_indices.size() > 0)
457  glDrawElements(m_primitives, (GLsizei) m_indices.size(), GL_UNSIGNED_INT, 0);
458  else
459  glDrawArrays(m_primitives, 0, (GLsizei) m_positions.size());
460 }
